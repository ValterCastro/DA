29void UFDS::linkSets(unsigned int i, unsigned int j) {
30 if (!isSameSet(i, j)) {
31 unsigned long x = findSet(i), y = findSet(j);
32 if (rank[x] > rank[y]) path[y] = x; // x becomes the root due to having a larger rank
33 else {
34 path[x] = y; // y becomes the root due to having a larger rank, or ...
35 if (rank[x] == rank[y]) rank[y]++; // ... due to both nodes having the same rank (in order to break the tie)

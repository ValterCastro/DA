31double haversine(Vertex *v1, Vertex *v2) {

33 double lonDiff = (v2->getLon() - v1->getLon()) * M_PI / 180;

35 double a = sin(latDiff / 2.00) * sin(latDiff/2.00) +
36 cos(v1->getLat() * M_PI / 180) * cos(v2->getLat() * M_PI / 180) *
37 sin(lonDiff/2.00) *sin(lonDiff/2.00);

35/*
36 * Auxiliary function to remove an outgoing edge (with a given destination (d))
37 * from a vertex (this).
38 * Returns true if successful, and false if such edge does not exist.
39 */
40bool Vertex::removeEdge(int id) {
41 bool removedEdge = false;
42 auto it = adj.begin();
43 while (it != adj.end()) {
44 Edge *edge = *it;
45 Vertex *dest = edge->getDest();
46 if (dest->getId() == id) {
47 it = adj.erase(it);
48 // Also remove the corresponding edge from the incoming list
49 auto it2 = dest->incoming.begin();
50 while (it2 != dest->incoming.end()) {
51 if ((*it2)->getOrig()->getId() == id) {
52 it2 = dest->incoming.erase(it2);
53 }
54 else {
55 it2++;
56 }
57 }
58 delete edge;
59 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
60 }
61 else {
62 it++;
63 }
64 }
65 return removedEdge;
66}

121double Graph::calculate_tsp(int node, int edge, std::vector<std::vector<double>>& dists) {
122 if (dists[node][edge] != -1) return dists[node][edge];
123
124 double ans = std::numeric_limits<double>::max();
125
126 if (edge == (1 << getNumVertex()) - 1) {
127 return findOrCalculateDistance(node, 0);
128 }
129
130 for (int i = 0; i < getNumVertex(); ++i) {
131 if (!(edge & (1 << i))) {
132 double tmp = findOrCalculateDistance(node, i) + calculate_tsp(i, edge | (1 << i), dists);
133 ans = std::min(ans, tmp);
134 }
135 }
136
137 return dists[node][edge] = ans;
138}

297double Graph::haversine(double lat1, double lon1, double lat2, double lon2){
298
299 double lat1Rad = lat1 * M_PI /180;
300 double lon1Rad = lon1 * M_PI /180;
301 double lat2Rad = lat2 * M_PI /180;
302 double lon2Rad = lon2 * M_PI /180;
303
304 double latDiff = lat2Rad -lat1Rad;
305 double lonDiff = lon2Rad -lon1Rad;
306
307 double a = sin(latDiff/2.00) * sin(latDiff/2.00) + cos(lat1Rad) * cos(lat2Rad) * sin(lonDiff/2.00) * sin(lonDiff/2.00);
308
309 double c = 2.00 * atan2(sqrt(a), sqrt(1-a));
310
311 double distance = 6371000.0 * c;
312
313 return distance;
314}
